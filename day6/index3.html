Câu lệnh WHERE
    câu lệnh WHERE để lọc dữ liệu từ bảng
    VD 1:
    SELECT * FROM customers WHERE id = 5;
    ( Chú ý: nếu WHERE cột  là số thì không cần để trong dấu nháy )
    VD 2:
    SELECT * FROM customers WHERE company = 'Company A';
    ( Chú ý : nếu WHERE cột là chuỗi thì phải đặt chuỗi trong dấu '' hay "" )
    

    các từ khóa AND OR NOT trong mệnh đề WHERE
    - AND ( phải đáp ứng tất cả các điều kiện )
        VD: lấy ra tất cả các khách hàng đến từ thành phố miami và có tên là Peter
            SELECT * FROM customers WHERE city = 'Miami' AND first_name = 'Peter'
    - OR ( chỉ cần đáp ứng 1 trong các điều kiện )
        VD: SELECT * FROM customers WHERE city = 'Miami' OR first_name = 'Anna'
            lấy ra tất cả các khách thoả mãn 1 trong 2 điều kiện ( đến từ Miami hoặc có tên là Anna )
    - NOT ( phủ định lại )
        VD: SELECT * FROM customers WHERE NOT city = 'Miami'
            lấy ra tất cả các khách hàng không đến từ Miami
