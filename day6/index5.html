Câu lệnh ORDER BY (Sắp xếp)
    ASC tăng dần
    DESC giảm dần
    Ví dụ :
    Sắp xếp khách hàng theo thứ tự id tăng dần
        SELECT * FROM customers
        ORDER BY id ASC;
    Ví dụ:
    Sắp xếp khách hàng theo thứ tự id giảm dần
        SELECT * FROM customers
        ORDER BY id DESC;
    Ví dụ:
    Sắp xếp nhiều cột
    Sắp xếp các khách hàng theo tên thành phố tăng dần A - Z
    và sắp xếp chức vụ giảm dần từ Z - A
        SELECT id,city,job_title FROM customers
        ORDER BY city ASC, job_title DESC;
